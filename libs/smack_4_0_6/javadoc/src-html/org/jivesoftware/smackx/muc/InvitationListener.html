018package org.jivesoftware.smackx.muc;
019
020import org.jivesoftware.smack.XMPPConnection;
021import org.jivesoftware.smack.packet.Message;
022
023/**
024 * A listener that is fired anytime an invitation to join a MUC room is received.
025 * 
026 * @author Gaston Dombiak
027 */
028public interface InvitationListener {
029
030    /**
031     * Called when the an invitation to join a MUC room is received.<p>
032     * 
033     * If the room is password-protected, the invitee will receive a password to use to join
034     * the room. If the room is members-only, the the invitee may be added to the member list.
035     * 
036     * @param conn the XMPPConnection that received the invitation.
037     * @param room the room that invitation refers to.
038     * @param inviter the inviter that sent the invitation. (e.g. crone1@shakespeare.lit).
039     * @param reason the reason why the inviter sent the invitation.
040     * @param password the password to use when joining the room.
041     * @param message the message used by the inviter to send the invitation.
042     */
043    public abstract void invitationReceived(XMPPConnection conn, String room, String inviter, String reason,
044                                            String password, Message message);
045
046}
