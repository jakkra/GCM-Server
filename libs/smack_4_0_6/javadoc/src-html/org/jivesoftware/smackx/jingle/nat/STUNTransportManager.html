017package org.jivesoftware.smackx.jingle.nat;
018
019import org.jivesoftware.smackx.jingle.JingleSession;
020
021/**
022 * A Jingle Transport Manager implementation to be used on NAT networks with  STUN Service NOT Blocked.
023 *
024 * @author Thiago Camargo
025 */
026public class STUNTransportManager extends JingleTransportManager {
027    STUNResolver stunResolver = null;
028
029    public STUNTransportManager() {
030        stunResolver = new STUNResolver() {
031        };
032        try {
033            stunResolver.initializeAndWait();
034        } catch (Exception e) {
035            e.printStackTrace();
036        }
037    }
038
039    protected TransportResolver createResolver(JingleSession session) {
040        try {
041            stunResolver.resolve(session);
042        } catch (Exception e) {
043            e.printStackTrace();
044        }
045        return stunResolver;
046    }
047}
