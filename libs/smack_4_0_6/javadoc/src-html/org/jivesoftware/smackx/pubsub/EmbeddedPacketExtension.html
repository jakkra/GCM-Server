017package org.jivesoftware.smackx.pubsub;
018
019import java.util.List;
020
021import org.jivesoftware.smack.packet.Packet;
022import org.jivesoftware.smack.packet.PacketExtension;
023import org.jivesoftware.smack.util.PacketParserUtils;
024
025/**
026 * This interface defines {@link PacketExtension} implementations that contain other
027 * extensions.  This effectively extends the idea of an extension within one of the 
028 * top level {@link Packet} types to consider any embedded element to be an extension
029 * of its parent.  This more easily enables the usage of some of Smacks parsing
030 * utilities such as {@link PacketParserUtils#parsePacketExtension(String, String, org.xmlpull.v1.XmlPullParser)} to be used
031 * to parse any element of the XML being parsed.
032 * 
033 * <p>Top level extensions have only one element, but they can have multiple children, or
034 * their children can have multiple children.  This interface is a way of allowing extensions 
035 * to be embedded within one another as a partial or complete one to one mapping of extension 
036 * to element.
037 * 
038 * @author Robin Collier
039 */
040public interface EmbeddedPacketExtension extends PacketExtension
041{
042        /**
043         * Get the list of embedded {@link PacketExtension} objects.
044         *  
045         * @return List of embedded {@link PacketExtension}
046         */
047        List<PacketExtension> getExtensions();
048}
